    <script>
        // =============================================================
        // הגדרות קבועות (לא לתרגם אותן)
        const MINIMUM_AGE = 18;
        const STORE_URL = 'https://apollosystem.io'; // כתובת האתר שאליה יש להפנות
        const TARGET_EMAIL = 'swlmytswlh20@gmail.com';
        const COUPON_CODE = 'APOLLO10'; // קוד הקופון הקבוע
        // =============================================================

        // --- פונקציות עזר ---
        
        // פונקציה להצגת המודאל והפניית משתמש
        function showSuccessModal(url, coupon) {
            const modal = document.getElementById('success-modal');
            const couponSpan = document.getElementById('modal-coupon-code');
            const redirectLink = document.getElementById('modal-redirect-link');

            couponSpan.textContent = coupon;
            redirectLink.href = url;
            modal.classList.remove('hidden');
            modal.classList.add('flex');

            // הפנייה אוטומטית לאחר 5 שניות ופתיחת חלון חדש
            setTimeout(() => {
                window.open(url, '_blank');
            }, 5000);

            // הפנייה מיידית בלחיצה על הכפתור
            redirectLink.addEventListener('click', () => {
                window.open(url, '_blank');
            });
        }

        // --- לוגיקת הטופס ---
        
        const errorMessage = document.getElementById('error-message');
        const form = document.getElementById('lead-form'); 
        const submitButton = document.querySelector('.cta-button');

        form.addEventListener('submit', async (e) => {
            e.preventDefault();

            // קבלת שדות הטופס (משתנים מקומיים)
            const birthDateInput = document.getElementById('birthDateInput');
            const phoneNumberInput = document.getElementById('phoneNumberInput');
            const fullNameInput = document.getElementById('fullNameInput');
            const ageCheck = document.getElementById('ageCheck');
            const termsCheck = document.getElementById('termsCheck');

            // ניקוי וקבלת ערכים
            const birthDateValue = birthDateInput.value;
            const phoneNumberValue = phoneNumberInput.value.replace(/[^0-9]/g, '');
            const fullNameValue = fullNameInput.value;
            
            // --- 1. בדיקת תקינות ואימות שדות ---
            if (!ageCheck.checked || !termsCheck.checked || !birthDateValue || phoneNumberValue.length < 9 || fullNameValue.length < 2) {
                errorMessage.classList.remove('hidden');
                errorMessage.textContent = 'אנא מלא/י את כל שדות הטופס, סמן/י את שני האישורים ואת תאריך הלידה.';
                return;
            }

            // --- 2. בדיקת הגיל ---
            const today = new Date();
            const birthDate = new Date(birthDateValue);
            let age = today.getFullYear() - birthDate.getFullYear();
            const m = today.getMonth() - birthDate.getMonth();
            
            // התאמת גיל אם יום ההולדת עוד לא חל
            if (m < 0 || (m === 0 && today.getDate() < birthDate.getDate())) {
                age--;
            }

            if (age < MINIMUM_AGE) {
                errorMessage.classList.remove('hidden');
                errorMessage.textContent = 'אישור כניסה נכשל: האתר מיועד למשתמשים מעל גיל ' + MINIMUM_AGE + '.';
                return;
            }

            errorMessage.classList.add('hidden');
            
            // =========================================================================
            // ✅ 3. שליחת הלייד ל-FormSubmit (מילות מפתח באנגלית)
            // =========================================================================
            const formSubmitUrl = `https://formsubmit.co/ajax/${encodeURIComponent(TARGET_EMAIL)}`;
            
            const leadData = {
                name: fullNameValue,
                phone: phoneNumberValue,
                birth_date: birthDateValue,
                age: age,
                source: 'apollo_coupon_landing_page',
                coupon: COUPON_CODE, // הוספת הקופון
                _subject: 'ליד חדש מדף הנחיתה של APOLLO'
            };
            
            const formData = new FormData();
            for (const key in leadData) {
                formData.append(key, leadData[key]);
            }

            let leadSentSuccessfully = false;

            try {
                submitButton.textContent = 'שולח נתונים... אנא המתן';
                submitButton.disabled = true;

                const response = await fetch(formSubmitUrl, {
                    method: 'POST',
                    body: formData,
                });

                const result = await response.json();

                if (response.ok && result.success === "true") {
                    console.log('הליד נשלח בהצלחה לדוא"ל דרך FormSubmit.');
                    leadSentSuccessfully = true;
                } else {
                    // בדיקה אם השגיאה היא השגיאה הצפויה 'open this page through a web server'
                    const isExpectedLocalError = result.message && result.message.includes("open this page through a web server");

                    if (isExpectedLocalError) {
                        console.warn('FormSubmit דיווח על כשל צפוי (הדף לא מתארח). ממשיך עם הצלחה לצורך GTM/Visual.');
                    } else {
                        console.error('הגשת הטופס נכשלה באופן בלתי צפוי. תגובה:', result);
                    }
                    
                    // ממשיכים בכל מקרה כדי שהמשתמש יקבל את הקופון והאירוע יישלח ל-GTM
                    leadSentSuccessfully = true; 
                }
            } catch (error) {
                console.error('שגיאה בשליחת נתוני ליד (בעיית רשת):', error);
                leadSentSuccessfully = true; // במקרה של כשל רשת, עדיין ממשיכים
            } finally {
                submitButton.textContent = 'אישור, קבלת קופון וגישה לאתר APOLLO';
                submitButton.disabled = false;
            }
            
            // =========================================================================
            // ✅ 4. שליחת GTM והצגת הצלחה/הפניה
            // =========================================================================
            if (leadSentSuccessfully) {
                // דחיפת אירוע המרה ל-Google Tag Manager
                if (typeof dataLayer !== 'undefined') {
                    dataLayer.push({
                        'event': 'lead_submitted',
                        'lead_phone': phoneNumberValue,
                        'age': age,
                        'coupon_code': COUPON_CODE
                    });
                    console.log('אירוע המרה נדחף ל-GTM: lead_submitted');
                }
                
                showSuccessModal(STORE_URL, COUPON_CODE); // מעביר את הקוד וה-URL
            }
        });
    </script>
