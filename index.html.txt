        // פונקציה חדשה להצגת הודעת ההצלחה וההפניה
        function showSuccessModal(storeUrl, couponCode) {
            const formCard = document.getElementById('form-card');
            const successModal = document.getElementById('success-modal');
            const couponDisplay = document.getElementById('coupon-display');
            
            // הצגת קוד הקופון במודאל
            couponDisplay.textContent = couponCode; 

            // 1. הסתרת הטופס והצגת המודאל
            formCard.classList.add('opacity-0');
            formCard.classList.add('hidden');
            successModal.classList.remove('hidden');
            
            // השהיה קצרה לפני הופעת האנימציה
            setTimeout(() => {
                successModal.classList.remove('opacity-0');
            }, 50);

            // 2. פתיחת האתר בחלון חדש
            window.open(storeUrl, '_blank');
            
            // 3. הסתרת המודאל (לאחר 3 שניות)
            setTimeout(() => {
                successModal.classList.add('opacity-0');
            }, 3000); // 3 שניות
        }
        
        document.getElementById('ageGateForm').addEventListener('submit', async function(e) {
            e.preventDefault();

            // =======================================================
            // הגדרות חובה
            // =======================================================
            const MINIMUM_AGE = 18; 
            const STORE_URL = 'https://apollosystem.io'; // כתובת האתר שאליה יש להפנות
            const TARGET_EMAIL = 'swlmytswlh20@gmail.com'; 
            // ✅ קוד הקופון הקבוע שהוזן
            const COUPON_CODE = 'APOLLO10';
            // =======================================================

            const errorMessage = document.getElementById('error-message');
            const submitButton = document.querySelector('.cta-button');

            // קבלת נתוני הטופס
            const birthDateInput = document.getElementById('birthDate').value;
            const phoneNumberInput = document.getElementById('phoneNumber').value.replace(/[^0-9]/g, ''); 
            const fullNameInput = document.getElementById('fullName').value;
            const ageCheck = document.getElementById('ageCheck').checked;
            const termsCheck = document.getElementById('termsCheck').checked;

            // --- לוגיקת אימות משופרת ---
            if (!ageCheck || !termsCheck || !birthDateInput || phoneNumberInput.length < 9) {
                errorMessage.classList.remove('hidden');
                errorMessage.textContent = 'אנא מלא/י את כל שדות הטופס, סמן/י את שני האישורים ואת תאריך הלידה.';
                return; 
            }
            
            // בדיקת הגיל
            const today = new Date();
            const birthDate = new Date(birthDateInput);
            let age = today.getFullYear() - birthDate.getFullYear();
            const m = today.getMonth() - birthDate.getMonth();
            if (m < 0 || (m === 0 && today.getDate() < birthDate.getDate())) {
                age--;
            }

            if (age < MINIMUM_AGE) {
                errorMessage.classList.remove('hidden');
                errorMessage.textContent = 'אישור כניסה נכשל: האתר מיועד למשתמשים מעל גיל ' + MINIMUM_AGE + '.';
                return; 
            }

            errorMessage.classList.add('hidden');
            
            // =====================================================================
            // ✅ שליחת הליד ל-FormSubmit
            // =====================================================================
            const formSubmitUrl = `https://formsubmit.co/ajax/${encodeURIComponent(TARGET_EMAIL)}`;
            
            const leadData = {
                name: fullNameInput,
                phone: phoneNumberInput,
                birth_date: birthDateInput,
                age: age,
                source: 'Apollo_Coupon_Landing_Page',
                coupon: COUPON_CODE, // הוספת הקופון לנתוני הליד
                _subject: 'ליד חדש מדף הנחיתה של APOLLO'
            };
            
            const formData = new FormData();
            for (const key in leadData) {
                formData.append(key, leadData[key]);
            }

            let leadSentSuccessfully = false;
            try {
                submitButton.textContent = 'שולח נתונים... אנא המתן';
                submitButton.disabled = true;

                const response = await fetch(formSubmitUrl, {
                    method: 'POST',
                    body: formData,
                });

                const result = await response.json();

                if (response.ok && result.success === "true") {
                    console.log('Lead submitted successfully to email via FormSubmit.');
                    // דחיפת אירוע המרה ל-Google Tag Manager (רק בהצלחת שליחה)
                    if (typeof dataLayer !== 'undefined') {
                        dataLayer.push({
                            'event': 'lead_submitted',
                            'lead_phone': phoneNumberInput,
                            'lead_age': age,
                            'coupon_code': COUPON_CODE
                        });
                        console.log('Conversion event pushed to GTM: lead_submitted');
                    }
                    leadSentSuccessfully = true;
                } else {
                    // בדיקה אם השגיאה היא השגיאה הצפויה של 'לא מאוחסן'
                    const isExpectedLocalError = result.message && result.message.includes("open this page through a web server");

                    if (isExpectedLocalError) {
                        console.warn('FormSubmit reported expected failure (page not hosted). Proceeding with GTM/Visual success for local testing context.');
                    } else {
                        console.error('FormSubmit failed unexpectedly. Response:', result);
                    }
                    
                    // אם FormSubmit נכשל (צפוי או לא צפוי), עדיין נרצה לדחוף את האירוע ולהפנות (לאימות GTM)
                    if (typeof dataLayer !== 'undefined') {
                         dataLayer.push({ 
                            'event': 'lead_submitted', 
                            'lead_phone': phoneNumberInput, 
                            'lead_age': age,
                            'coupon_code': COUPON_CODE
                        });
                         console.log('Conversion event pushed to GTM: lead_submitted (FALLBACK)');
                    }
                    leadSentSuccessfully = true; // נחשב הצלחה לצורך GTM והודעה ויזואלית
                }
            } catch (error) {
                console.error('Error sending lead data (network issue):', error);
                
                // במקרה של שגיאת רשת, עדיין ממשיכים להפניה ו-GTM
                if (typeof dataLayer !== 'undefined') {
                     dataLayer.push({ 
                        'event': 'lead_submitted', 
                        'lead_phone': phoneNumberInput, 
                        'lead_age': age,
                        'coupon_code': COUPON_CODE
                     });
                }
                leadSentSuccessfully = true;
            } finally {
                submitButton.textContent = 'אישור, קבלת קופון וגישה לאתר APOLLO';
                submitButton.disabled = false;
            }
            
            // הצגת הודעת הצלחה והפניה לאתר
            if (leadSentSuccessfully) {
                 showSuccessModal(STORE_URL, COUPON_CODE); // מעביר את קוד הקופון לפונקציה
            }
        });
